// hello.jsx

    import React from 'react';
    export default class Hello extends React.Component {
        render() {
            return <h1>Hello world</h1>;
        }
    }

// main.jsx

    import React from 'react';
    import ReactDOM from 'react-dom';
    import Hello from './hello.jsx';
    
    function main() {
        ReactDOM.render(
            <Hello />,
            document.getElementById('app')
        );
    }
    
    main();

// index.html

    <!DOCTYPE html>
